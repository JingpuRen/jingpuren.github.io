    public boolean check(int row,int col){
        for(int i = 0;i < n;i++){
            if(g[i][col] == 'Q') return false;
        }

        for(int i = row - 1,j = col - 1;i >= 0 && j >= 0;i--,j--){
            if(g[i][j] == 'Q') return false;
        }

        for(int i = row - 1,j = col + 1;i >= 0 && j < n;i--,j++){
            if(g[i][j] == 'Q') return false;
        }

        return true;
    }
